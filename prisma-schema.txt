// prisma/schema.prisma
datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

generator client {
  provider = "prisma-client-js"
}

// ============ Users & Auth ============
model User {
  id                String      @id @default(cuid())
  email             String      @unique
  name              String?
  oauthProvider     String?     // "google", "github", "microsoft"
  oauthId           String?
  passwordHash      String?     // For custom auth
  emailVerified     Boolean     @default(false)
  lastLogin         DateTime?
  createdAt         DateTime    @default(now())
  updatedAt         DateTime    @updatedAt

  // Relations
  userRoles         UserRole[]
  courses           Course[]    @relation("createdBy")
  auditLogs         AuditLog[]
  uploadedFiles     UploadedFile[]

  @@unique([oauthProvider, oauthId])
  @@index([email])
}

model Role {
  id                String      @id @default(cuid())
  name              String      @unique // "admin", "instructor", "student", "viewer"
  description       String?
  createdAt         DateTime    @default(now())

  // Relations
  userRoles         UserRole[]
  permissions       Permission[]

  @@index([name])
}

model Permission {
  id                String      @id @default(cuid())
  name              String      @unique // "create:course", "read:course", "delete:user"
  description       String?

  // Relations
  roles             Role[]

  @@index([name])
}

model UserRole {
  id                String      @id @default(cuid())
  userId            String
  roleId            String
  courseId          String?     // Optional: course-level role
  createdAt         DateTime    @default(now())

  // Relations
  user              User        @relation(fields: [userId], references: [id], onDelete: Cascade)
  role              Role        @relation(fields: [roleId], references: [id], onDelete: Cascade)
  course            Course?     @relation(fields: [courseId], references: [id], onDelete: SetNull)

  @@unique([userId, roleId, courseId])
  @@index([userId])
  @@index([roleId])
  @@index([courseId])
}

// ============ Content ============
model Course {
  id                String      @id @default(cuid())
  titleEs           String
  titleEn           String
  descriptionEs     String?
  descriptionEn     String?
  createdById       String
  price             Float       @default(0)
  currency          String      @default("USD")
  collaborationStatus String    @default("open") // "open", "closed", "archived"
  translationStatusEs String    @default("master") // language version status
  translationStatusEn String    @default("auto-draft")
  isPublished       Boolean     @default(false)
  metadata          Json?       // Additional data
  createdAt         DateTime    @default(now())
  updatedAt         DateTime    @updatedAt

  // Relations
  createdBy         User        @relation("createdBy", fields: [createdById], references: [id])
  chapters          Chapter[]
  userRoles         UserRole[]

  @@index([createdById])
  @@index([isPublished])
}

model Chapter {
  id                String      @id @default(cuid())
  courseId          String
  titleEs           String
  titleEn           String
  summaryEs         String?
  summaryEn         String?
  orderIndex        Int
  contentJson       Json?       // Generated by ETL pipeline
  keywords          String[]    @default([])
  createdAt         DateTime    @default(now())
  updatedAt         DateTime    @updatedAt

  // Relations
  course            Course      @relation(fields: [courseId], references: [id], onDelete: Cascade)
  figures           Figure[]
  codeSnippets      CodeSnippet[]

  @@unique([courseId, orderIndex])
  @@index([courseId])
}

model Figure {
  id                String      @id @default(cuid())
  chapterId         String
  s3Key             String      // S3 path to image
  captionEs         String?
  captionEn         String?
  altTextEs         String?
  altTextEn         String?
  fileType          String      // "png", "jpg", etc.
  fileSizeBytes     Int
  createdAt         DateTime    @default(now())

  // Relations
  chapter           Chapter     @relation(fields: [chapterId], references: [id], onDelete: Cascade)

  @@index([chapterId])
}

model CodeSnippet {
  id                String      @id @default(cuid())
  chapterId         String
  language          String      // "python", "javascript", etc.
  s3Key             String      // S3 path to code file
  descriptionEs     String?
  descriptionEn     String?
  lineCount         Int?
  createdAt         DateTime    @default(now())

  // Relations
  chapter           Chapter     @relation(fields: [chapterId], references: [id], onDelete: Cascade)

  @@index([chapterId])
  @@index([language])
}

// ============ File Uploads ============
model UploadedFile {
  id                String      @id @default(cuid())
  userId            String
  originalName      String
  s3Key             String      @unique
  fileType          String
  fileSizeBytes     Int
  checksum          String      // SHA-256 for integrity
  scanStatus        String      @default("pending") // "pending", "clean", "infected"
  externalUrl       String?     // Presigned URL for download
  expiresAt         DateTime?   // When presigned URL expires
  isPublic          Boolean     @default(false)
  createdAt         DateTime    @default(now())
  deletedAt         DateTime?

  // Relations
  uploadedBy        User        @relation(fields: [userId], references: [id])

  @@index([userId])
  @@index([createdAt])
}

// ============ Audit & Security ============
model AuditLog {
  id                String      @id @default(cuid())
  userId            String?
  action            String      // "create:course", "read:file", etc.
  resourceType      String      // "course", "user", "file"
  resourceId        String?
  status            String      // "success", "failure"
  ipAddress         String?
  userAgent         String?
  details           Json?       // Additional context
  timestamp         DateTime    @default(now())

  // Relations
  user              User?       @relation(fields: [userId], references: [id])

  @@index([userId])
  @@index([action])
  @@index([timestamp])
  @@index([resourceType])
}

model ApiKey {
  id                String      @id @default(cuid())
  name              String
  key               String      @unique      // Hashed
  lastUsed          DateTime?
  rateLimit         Int         @default(1000) // Requests per hour
  isActive          Boolean     @default(true)
  createdAt         DateTime    @default(now())
  expiresAt         DateTime?

  @@index([isActive])
}
